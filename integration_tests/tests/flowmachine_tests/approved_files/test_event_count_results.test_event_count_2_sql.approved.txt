SELECT subscriber,
       count(*) AS value
FROM (SELECT events.calls.msisdn AS subscriber
      FROM events.calls
      WHERE events.calls.datetime >= '2016-01-05 00:00:00'

      UNION ALL

      SELECT events.sms.msisdn AS subscriber
      FROM events.sms
      WHERE events.sms.datetime >= '2016-01-05 00:00:00') AS u
GROUP BY subscriber
