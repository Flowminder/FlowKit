SELECT subscriber,
       count(*) AS value
FROM (SELECT events.calls.msisdn AS subscriber
      FROM events.calls
      WHERE (events.calls.datetime >= '2016-01-02 00:00:00')
        AND (events.calls.datetime < '2016-01-04 00:00:00')
        AND EXTRACT(HOUR FROM events.calls.datetime) BETWEEN 12 AND 17

      UNION ALL

      SELECT events.sms.msisdn AS subscriber
      FROM events.sms
      WHERE (events.sms.datetime >= '2016-01-02 00:00:00')
        AND (events.sms.datetime < '2016-01-04 00:00:00')
        AND EXTRACT(HOUR FROM events.sms.datetime) BETWEEN 12 AND 17) AS u
GROUP BY subscriber
